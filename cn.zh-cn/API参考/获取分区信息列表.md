    ```
    NONE
    ```

    ```
    {
        "is_success": true,
        "message": "list partitions succeed",
        "partitions": {
            "total_count": 7,
            "partition_infos": [
                {
                    "partition_name": "name=test",
                    "create_time": 1579520179000,
                    "last_access_time": 1579520179000,
                    "locations": [
                        "s3a://obs-jptest/partition"
                    ]
                },
    ```
